*& Form zf_atualiza

FORM zf_atualiza .

  SELECT * FROM vbap
    INTO TABLE @DATA(lt_vbap)
    WHERE erdat = @sy-datum.

    LOOP AT lt_vbap INTO DATA(ls_vbap).

      READ TABLE lt_ztdsr01 INTO ls_ztdsr01 WITH KEY uname = ls_vbap-ernam.

      ls_vbap-erdat = ls_vbap-erdat + 1.
      ls_vbap-ernam = ls_ztdsr01-name.

      MODIFY lt_vbap FROM ls_vbap INDEX sy-tabix.
    ENDLOOP.

*  MODIFY vbap FROM TABLE lt_vbap.

ENDFORM.

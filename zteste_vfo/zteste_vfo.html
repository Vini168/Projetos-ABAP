  TYPES:BEGIN OF ty_vendas,
          vbeln TYPE vbap-vbeln,
*          posnr TYPE vbap-posnr,
*          matnr TYPE vbap-matnr,
          knumv TYPE vbak-knumv,
        END OF ty_vendas.

  DATA:  tg_vbap TYPE TABLE OF vbap,
         wa_vbap TYPE ty_vendas.

  PARAMETER p_vbeln TYPE vbap-vbeln.

  IF p_vbeln IS NOT INITIAL.

    SELECT
      vbap~vbeln
*      vbap~posnr
*      vbap~matnr

    FROM vbap
    INNER JOIN vbak ON vbak~vbeln = vbap~vbeln
    INTO TABLE tg_vbap
    WHERE vbap~vbeln = p_vbeln.

    LOOP AT tg_vbap INTO  wa_vbap.
      WRITE:
      sy-vline, wa_vbap-vbeln,/,

      sy-vline, wa_vbap-knumv, /.

    ENDLOOP.
